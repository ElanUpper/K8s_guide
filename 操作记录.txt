-- 回退deploy
kubectl rollout status deploy kubernetes-bootcamp


-- 通过yaml方式创建pod/deployment
kubectl create -f *.yaml
  -- 创建pods yaml
    apiVersion: v1
    kind: Pod
    metadata:
      name: nginx
    spec:
      containers:
        - name: nginx
          image: nginx:1.7.9
          ports:
          - containerPort: 80
              
  -- 创建deployment yaml
    apiVersion: apps/v1beta1
    kind: Deployment
    metadata:
      name: nginx-deploy
    spec:
      replicas: 2
      template:
        metadata:
          labels: 
            app: nginx
        spec:
          containers:
          - name: nginx
            image: nginx:1.7.9
            ports:
              - containerPort: 80
